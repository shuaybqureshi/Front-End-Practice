What is the issue with this code, if we expect the value of

sum(1, 2, 3, 4, 5, "a", NaN)

to be equal to 15?

function sum() {
      let answer = 0;
      arguments.forEach((element) => {
        const e = parseInt(element);
        if (!isNaN(e)) {
          answer += e;
        }
      });
      return answer;
    }
